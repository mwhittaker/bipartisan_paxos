\begin{algorithm}[ht]
  \caption{Majority Commit \BPaxos{} proposer modification to prevent deadlock.}%
  \algolabel{MajorityCommitBPaxosProposerModification}
  \begin{algorithmic}[1]
    \makeatletter
    \setcounter{ALG@line}{8}
    \makeatother
    \State $M \gets$ the set of acceptors in $A$ that voted for $v'$ in round $0$
           \linelabel{McbmM}
    \If{$\exists v_y \in \deps{v_x}$ such that no acceptor in $M$ knows that $v_y$ is chosen}
        \linelabel{McbmObliviousM}
      \State send any value satisfying the pruned dependency invariant
    \EndIf{}
    \If{the proposer was recovering $v_z$ and deferred to the recovery of $v_x$}
      \If{$v_z \in \deps{v_x}$}
        \linelabel{McbmVzinVx}
        \State abort recovery of $v_z$; $v_z$ has already been chosen
      \Else{}
        \linelabel{McbmVzNotinVx}
        \State in vertex $v_z$, send any value satisfying
               \invref{PrunedDependencies}
      \EndIf{}
    \EndIf{}
  \end{algorithmic}
\end{algorithm}
