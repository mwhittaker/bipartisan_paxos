\newcommand{\nullbot}{\textsf{null}}
\newcommand{\newstuff}[1]{\textcolor{red}{#1}}


\begin{algorithm}[ht]
  \caption{%
    Unanimous \BPaxos{} Proposer. Changes to \algoref{FastPaxosProposer} are
    underlined and highlighted in red.
    \TODO[mwhittaker]{This is confusing because $x$ and $y$ are of the form $(x, \deps{v_x})$. Fix that.}
  }%
  \algolabel{UnanimousBPaxosProposer}
  \begin{algorithmic}[1]
    \GlobalState a value $v$, initially \nullbot{}
    \GlobalState a round $i$, initially $-1$

    \Upon{%
      \newstuff{receiving \msg{Phase2B}{0, v'} from all $2f + 1$ acceptors as
      the proposer of round $0$ with $i=0$}
    }
      \If{every value of $v'$ is the same}\linelabel{UnanimousUnanimous}
        \State $v'$ is chosen, notify the learners
      \Else{}
        \State $i \gets 1$
        \State \newstuff{$v \gets$ an arbitrary value satisfying the dependency invariant}
        \State \newstuff{send \msg{Phase2a}{i, x} to at least $f + 1$ acceptors}
      \EndIf
    \EndUpon

    \Upon{recovery}
      \State $i \gets$ next largest round owned by this proposer
      \State send \msg{Phase1A}{i} to at least $f+1$ acceptors
    \EndUpon

    \Upon{receiving \msg{Phase1B}{i, vr, vv} from $f+1$ acceptors}
      \State $k \gets$ the largest $vr$ in any \msg{Phase1B}{i, vr, vv}
      \If{$k = -1$}
        \State $v \gets$ an arbitrary value satisfying the dependency invariant
      \ElsIf{$k > 0$}
        \State $v \gets$ the corresponding $vv$ in round $k$
      \ElsIf{$k = 0$}
      \If{\newstuff{all $f + 1$ messages are of the form} \msg{Phase1B}{i, 0, v'} for some value $v'$}\linelabel{UnanimousBPaxos}
        \State $v \gets$ $v'$ \linelabel{Unanimousk0Start}
        \Else{}
          \State $v \gets$ an arbitrary value satisfying the dependency invariant
        \EndIf
      \EndIf
      \State send \msg{Phase2a}{i, v} to at least $f + 1$ acceptors \linelabel{Unanimousk0End}
    \EndUpon

    \Upon{receiving \msg{Phase2B}{i} from $f+1$ acceptors}
      \State $v$ is chosen, notify the learners
    \EndUpon
  \end{algorithmic}
\end{algorithm}
